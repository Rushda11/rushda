#include <iostream>
using namespace std;

const int max = 5;
int queue_arr[max];
int front, rear = -1;

void enqueue(){
	int value;
	if (rear == max - 1) {
		cout << "queue flowerflow, cannot insert";
	}
	else {
		cout << "Enter element to insert:";
		cin >> value;
	}
	if (front == -1) {
		front = 0;
	}
	rear++;
	queue_arr[rear] = value;
	cout << value <<" "<< "inserted into the queue" << endl;
}

void dequeue() {
	if (front == -1 || front > rear) {
		cout << "queue underflow";
		front, rear = -1;
	}
	else {
		cout << "element deleated:" << queue_arr[front] << endl;
		front++;
	}
}

void display() {
	if (front == -1 || front > rear) {
		cout << "queue is empty" << endl;
	}
	else {
		cout << "queue elements"<<endl;
		for (int i = front; i<=rear; i++) {
			cout << queue_arr[i] <<"  ";
		}
		cout << endl;
	}
}

void peek() {
	cout << "first element of queue is:" << queue_arr[front]<<endl;
}

int main() {
	int ch, value;
	cout << "1)Add an element in the queue" << endl;
	cout << "2)Remove an element from queue" << endl;
	cout << "3)Peek queue" << endl;
	cout << "4)Display the queue" << endl;
	cout << "5)Quit" << endl;
	do {
		cout << endl << "enter your choice:";
		cin >> ch;
		switch (ch) {
		case 1:enqueue();
			break;
		case 2:dequeue();
			break;
		case 3:peek();
			break;
		case 4:display();
			break;
		case 5:break;
			break;
		default:cout << "invalid option";
		}
	} while (ch != 5);
}
